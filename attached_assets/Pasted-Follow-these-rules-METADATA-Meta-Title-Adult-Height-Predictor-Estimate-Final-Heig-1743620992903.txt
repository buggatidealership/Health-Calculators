        function toggleUnit(person, unit) {
            // Toggle buttons

            buttons.forEach(btn => {

        }
       
        function calculateHeight() {
            // Get input values
            const gender = document.getElementById('childGender').value;
            const age = parseFloat(document.getElementById('childAge').value);
           
            // Get child height
            let childHeight;
            if (document.getElementById('child-cm-input').style.display !== 'none') {
                childHeight = parseFloat(document.getElementById('childHeightCm').value);
            } else {
                const feet = parseFloat(document.getElementById('childHeightFt').value) || 0;
                const inches = parseFloat(document.getElementById('childHeightIn').value) || 0;
                childHeight = feet * 30.48 + inches * 2.54;
            }
           
            // Get mother's height
            let motherHeight;
            if (document.getElementById('mother-cm-input').style.display !== 'none') {
                motherHeight = parseFloat(document.getElementById('motherHeightCm').value);
            } else {
                const feet = parseFloat(document.getElementById('motherHeightFt').value) || 0;
                const inches = parseFloat(document.getElementById('motherHeightIn').value) || 0;
                motherHeight = feet * 30.48 + inches * 2.54;
            }
           
            // Get father's height
            let fatherHeight;
            if (document.getElementById('father-cm-input').style.display !== 'none') {
                fatherHeight = parseFloat(document.getElementById('fatherHeightCm').value);
            } else {
                const feet = parseFloat(document.getElementById('fatherHeightFt').value) || 0;
                const inches = parseFloat(document.getElementById('fatherHeightIn').value) || 0;
                fatherHeight = feet * 30.48 + inches * 2.54;
            }
           
            // Validate inputs
            if (isNaN(childHeight) || isNaN(motherHeight) || isNaN(fatherHeight)) {
                alert("Please enter valid height values");
                return;
            }
           
            // Calculate mid-parental height (Tanner method)
            let midParentalHeight;
            if (gender === 'male') {
                midParentalHeight = ((motherHeight + fatherHeight) / 2) + 6.5;
            } else {
                midParentalHeight = ((motherHeight + fatherHeight) / 2) - 6.5;
            }
           
            // Calculate predicted height based on current height and age
            let predictedHeight;
            let additionalInfo = "";
           
            if (age < 4) {
                predictedHeight = midParentalHeight;
                additionalInfo = "Predictions for children under 4 years are less reliable as growth patterns can vary significantly.";
            } else {
                const growthPercentile = getGrowthPercentile(gender, age, childHeight);
                const geneticWeight = 0.6;
                const currentWeight = 0.4;
               
                predictedHeight = (geneticWeight * midParentalHeight) + (currentWeight * projectHeight(gender, age, childHeight, growthPercentile));
               
                if (gender === 'male' && age > 12) {
                    predictedHeight *= 1.02;
                } else if (gender === 'female' && age > 10) {
                    predictedHeight *= 1.01;
                }
            }
           
            // Calculate range (±5 cm is typical margin of error)
            const lowerRange = predictedHeight - 5;
            const upperRange = predictedHeight + 5;
           
            // Display results in both units
            const feetInches = cmToFeetInches(predictedHeight);
            const lowerFeetInches = cmToFeetInches(lowerRange);
            const upperFeetInches = cmToFeetInches(upperRange);
           
            document.getElementById('predictedHeight').innerHTML =
                `${Math.round(predictedHeight)} cm (${feetInches})<br>
                Range: ${Math.round(lowerRange)}-${Math.round(upperRange)} cm (${lowerFeetInches}-${upperFeetInches})`;
            document.getElementById('additionalInfo').textContent = additionalInfo;
            document.getElementById('result').style.display = 'block';
        }
       
        function getGrowthPercentile(gender, age, height) {
            if (gender === 'male') {
                if (age < 10) {
                    if (height < 115) return 0.1;
                    if (height < 125) return 0.25;
                    if (height < 135) return 0.5;
                    if (height < 145) return 0.75;
                    return 0.9;
                } else {
                    if (height < 140) return 0.1;
                    if (height < 150) return 0.25;
                    if (height < 160) return 0.5;
                    if (height < 170) return 0.75;
                    return 0.9;
                }
            } else {
                if (age < 10) {
                    if (height < 110) return 0.1;
                    if (height < 120) return 0.25;
                    if (height < 130) return 0.5;
                    if (height < 140) return 0.75;
                    return 0.9;
                } else {
                    if (height < 135) return 0.1;
                    if (height < 145) return 0.25;
                    if (height < 155) return 0.5;
                    if (height < 165) return 0.75;
                    return 0.9;
                }
            }
        }
       
        function projectHeight(gender, age, height, percentile) {
            const remainingGrowth = gender === 'male' ?
                (18 - age) * (2.5 + (1 - percentile) * 1.5) :
                (16 - age) * (2.0 + (1 - percentile) * 1.5);
           
            return height + remainingGrowth;
        }
       
        function cmToFeetInches(cm) {
            const inches = cm / 2.54;
            const feet = Math.floor(inches / 12);
            const remainingInches = Math.round(inches % 12 * 10) / 10;
            return `${feet}'${remainingInches}"`;
        }
