        function toggleUnits() {
            const metricChecked = document.getElementById('metric').checked;
            document.querySelector('.metric-inputs').style.display = metricChecked ? 'block' : 'none';
            document.querySelector('.imperial-inputs').style.display = metricChecked ? 'none' : 'block';
        }

        function calculate() {
            // Reset warnings
            document.getElementById('age-warning').style.display = 'none';
           
            // Get age
            const ageYears = parseInt(document.getElementById('age-years').value) || 0;
            const ageMonths = parseInt(document.getElementById('age-months').value) || 0;
            const totalMonths = ageYears * 12 + ageMonths;
           
            // Validate age
            if (totalMonths > 240) { // 20 years
                document.getElementById('age-warning').textContent = 'Note: This calculator is designed for children up to 19 years. Results may not be accurate for older individuals.';
                document.getElementById('age-warning').style.display = 'block';
            }
           
            // Get height and weight based on selected units
            let heightCm, weightKg;
           
            if (document.getElementById('metric').checked) {
                heightCm = parseFloat(document.getElementById('height-cm').value);
                weightKg = parseFloat(document.getElementById('weight-kg').value);
            } else {
                const heightFt = parseFloat(document.getElementById('height-ft').value) || 0;
                const heightIn = parseFloat(document.getElementById('height-in').value) || 0;
                heightCm = (heightFt * 12 + heightIn) * 2.54;
               
                const weightLbs = parseFloat(document.getElementById('weight-lbs').value);
                weightKg = weightLbs * 0.453592;
            }
           
            // Simple validation
            if (!totalMonths || !heightCm || !weightKg) {
                alert("Please fill in all fields with valid numbers");
                return;
            }
           
            // SIMPLIFIED CALCULATIONS (for demo only)
            // These are made-up numbers just to show the concept
            const heightPercentile = Math.min(99, Math.max(1, 50 + (heightCm - (80 + totalMonths * 0.5))));
            const weightPercentile = Math.min(99, Math.max(1, 50 + (weightKg - (10 + totalMonths * 0.1)) * 2));
            const bmi = weightKg / ((heightCm/100) ** 2);
            const bmiPercentile = Math.min(99, Math.max(1, 50 + (bmi - 16) * 3));
           
            // Update visual bars
            document.getElementById('height-bar').style.width = `${heightPercentile}%`;
            document.getElementById('weight-bar').style.width = `${weightPercentile}%`;
            document.getElementById('bmi-bar').style.width = `${bmiPercentile}%`;
           
            // Show results
            document.getElementById('heightResult').textContent = `${heightCm.toFixed(1)} cm (${heightPercentile.toFixed(0)}th percentile)`;
            document.getElementById('weightResult').textContent = `${weightKg.toFixed(1)} kg (${weightPercentile.toFixed(0)}th percentile)`;
            document.getElementById('bmiResult').textContent = `BMI: ${bmi.toFixed(1)} (${bmiPercentile.toFixed(0)}th percentile)`;
           
            // Generate interpretations
            interpretHeight(heightPercentile);
            interpretWeight(weightPercentile);
            interpretBMI(bmiPercentile, ageYears);
           
            // Show results section
            document.getElementById('results').style.display = 'block';
        }
       
        function interpretHeight(percentile) {
            const elem = document.getElementById('height-interpretation');
            elem.className = 'interpretation';
           
            if (percentile < 3) {
                elem.innerHTML = `<strong>Very Short Stature:</strong> Your child's height is below the 3rd percentile. This may indicate growth issues. We recommend consulting with a pediatrician to evaluate potential causes.`;
                elem.className += ' alert';
            } else if (percentile < 10) {
                elem.innerHTML = `<strong>Short Stature:</strong> Your child's height is between the 3rd and 10th percentiles. While this may be normal for your family, consider discussing with your pediatrician at the next visit.`;
                elem.className += ' warning';
            } else if (percentile < 25) {
                elem.innerHTML = `<strong>Below Average Height:</strong> Your child's height is between the 10th and 25th percentiles. This is within the normal range but on the shorter side.`;
            } else if (percentile < 75) {
                elem.innerHTML = `<strong>Average Height:</strong> Your child's height is between the 25th and 75th percentiles, which is typical for their age and gender.`;
            } else if (percentile < 90) {
                elem.innerHTML = `<strong>Above Average Height:</strong> Your child's height is between the 75th and 90th percentiles. This is within the normal range but on the taller side.`;
            } else if (percentile < 97) {
                elem.innerHTML = `<strong>Tall Stature:</strong> Your child's height is between the 90th and 97th percentiles. While often normal, extremely rapid growth should be evaluated.`;
                elem.className += ' warning';
            } else {
                elem.innerHTML = `<strong>Very Tall Stature:</strong> Your child's height is above the 97th percentile. This may be normal genetic variation or indicate hormonal issues. Consider consulting with a pediatrician.`;
                elem.className += ' alert';
            }
        }
       
        function interpretWeight(percentile) {
            const elem = document.getElementById('weight-interpretation');
            elem.className = 'interpretation';
           
            if (percentile < 5) {
                elem.innerHTML = `<strong>Underweight:</strong> Your child's weight is below the 5th percentile. This may indicate inadequate nutrition or other health issues. Please consult with your pediatrician.`;
                elem.className += ' alert';
            } else if (percentile < 25) {
                elem.innerHTML = `<strong>Below Average Weight:</strong> Your child's weight is between the 5th and 25th percentiles. This may be normal for their height and build.`;
            } else if (percentile < 75) {
                elem.innerHTML = `<strong>Average Weight:</strong> Your child's weight is between the 25th and 75th percentiles, which is typical for their age and gender.`;
            } else if (percentile < 85) {
                elem.innerHTML = `<strong>Above Average Weight:</strong> Your child's weight is between the 75th and 85th percentiles. Monitor to ensure healthy growth patterns continue.`;
                elem.className += ' warning';
            } else if (percentile < 95) {
                elem.innerHTML = `<strong>High Weight:</strong> Your child's weight is between the 85th and 95th percentiles. Consider discussing nutrition and activity levels with your pediatrician.`;
                elem.className += ' warning';
            } else {
                elem.innerHTML = `<strong>Very High Weight:</strong> Your child's weight is above the 95th percentile. Please consult with your pediatrician to evaluate for potential weight-related health concerns.`;
                elem.className += ' alert';
            }
        }
       
        function interpretBMI(percentile, ageYears) {
            const elem = document.getElementById('bmi-interpretation');
            const statusElem = document.getElementById('weight-status');
            elem.className = 'interpretation';
            statusElem.className = 'interpretation';
           
            if (percentile < 5) {
                elem.textContent = "BMI is below the 5th percentile for age and gender.";
                statusElem.innerHTML = `<strong>Weight Status: Underweight</strong> - This may indicate inadequate nutrition or other health concerns. Please consult with your pediatrician.`;
                statusElem.className += ' alert';
            } else if (percentile < 85) {
                elem.textContent = "BMI is between the 5th and 85th percentiles for age and gender.";
                statusElem.innerHTML = `<strong>Weight Status: Healthy Weight</strong> - Your child's weight is appropriate for their height. Continue with healthy eating and activity habits.`;
            } else if (percentile < 95) {
                elem.textContent = "BMI is between the 85th and 95th percentiles for age and gender.";
                if (ageYears >= 2) {
                    statusElem.innerHTML = `<strong>Weight Status: Overweight</strong> - Consider reviewing diet and physical activity. Consult your pediatrician for guidance on healthy growth.`;
                } else {
                    statusElem.innerHTML = `<strong>Weight Status: High Weight-for-Length</strong> - For children under 2, consult your pediatrician about appropriate feeding practices.`;
                }
                statusElem.className += ' warning';
            } else {
                elem.textContent = "BMI is at or above the 95th percentile for age and gender.";
                if (ageYears >= 2) {
                    statusElem.innerHTML = `<strong>Weight Status: Obese</strong> - Please consult with your pediatrician to develop a plan for healthy weight management.`;
                } else {
                    statusElem.innerHTML = `<strong>Weight Status: Very High Weight-for-Length</strong> - For children under 2, this requires pediatric evaluation of feeding practices and growth patterns.`;
                }
                statusElem.className += ' alert';
            }
        }
