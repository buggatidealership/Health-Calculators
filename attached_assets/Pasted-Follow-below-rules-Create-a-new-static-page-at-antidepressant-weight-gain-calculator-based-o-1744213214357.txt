function AntidepressantWeightCalculator() {
  const [weight, setWeight] = React.useState("");
  const [weightUnit, setWeightUnit] = React.useState("kg");
  const [medication, setMedication] = React.useState("");
  const [duration, setDuration] = React.useState("");
  const [result, setResult] = React.useState(null);
  
  // Simple medication data
  const medicationData = {
    mirtazapine: { name: "Mirtazapine (Remeron)", avgGain: 2.5, riskLevel: "High" },
    paroxetine: { name: "Paroxetine (Paxil)", avgGain: 1.7, riskLevel: "High" },
    amitriptyline: { name: "Amitriptyline (Elavil)", avgGain: 1.8, riskLevel: "High" },
    citalopram: { name: "Citalopram (Celexa)", avgGain: 1.1, riskLevel: "Moderate" },
    fluoxetine: { name: "Fluoxetine (Prozac)", avgGain: 0.8, riskLevel: "Low-Moderate" },
    bupropion: { name: "Bupropion (Wellbutrin)", avgGain: -0.4, riskLevel: "Minimal/Weight Loss" }
  };
  
  // Duration factors
  const durationFactors = {
    "1": 0.5,
    "6": 1.0,
    "12": 1.2
  };
  
  function handleCalculate() {
    const currentWeight = parseFloat(weight);
    
    if (isNaN(currentWeight) || !medication || !duration) {
      return;
    }
    
    // Convert to kg if needed
    const baseWeight = weightUnit === "lb" ? currentWeight * 0.453592 : currentWeight;
    
    // Get values
    const med = medicationData[medication];
    const durationFactor = durationFactors[duration];
    
    // Calculate weight change
    const changeInKg = (baseWeight * med.avgGain / 100) * durationFactor;
    const changeInUnit = weightUnit === "lb" ? changeInKg * 2.20462 : changeInKg;
    const newWeight = currentWeight + changeInUnit;
    
    setResult({
      medication: med.name,
      weightChange: changeInUnit.toFixed(1),
      newWeight: newWeight.toFixed(1),
      riskLevel: med.riskLevel,
      unit: weightUnit
    });
  }
  
  return (
    <div style={{ fontFamily: "Arial", maxWidth: "500px", margin: "0 auto", padding: "20px" }}>
      <h2 style={{ color: "#333" }}>Antidepressant Weight Impact Calculator</h2>
      
      <div style={{ marginBottom: "15px" }}>
        <label style={{ display: "block", marginBottom: "5px" }}>
          Weight:
          <input
            type="number"
            value={weight}
            onChange={(e) => setWeight(e.target.value)}
            style={{ width: "100px", padding: "5px", marginLeft: "10px" }}
          />
          <select 
            value={weightUnit}
            onChange={(e) => setWeightUnit(e.target.value)}
            style={{ marginLeft: "5px", padding: "5px" }}
          >
            <option value="kg">kg</option>
            <option value="lb">lb</option>
          </select>
        </label>
      </div>
      
      <div style={{ marginBottom: "15px" }}>
        <label style={{ display: "block", marginBottom: "5px" }}>
          Medication:
          <select
            value={medication}
            onChange={(e) => setMedication(e.target.value)}
            style={{ width: "200px", padding: "5px", marginLeft: "10px" }}
          >
            <option value="">Select...</option>
            {Object.keys(medicationData).map(key => (
              <option key={key} value={key}>
                {medicationData[key].name}
              </option>
            ))}
          </select>
        </label>
      </div>
      
      <div style={{ marginBottom: "15px" }}>
        <label style={{ display: "block", marginBottom: "5px" }}>
          Duration:
          <select
            value={duration}
            onChange={(e) => setDuration(e.target.value)}
            style={{ width: "150px", padding: "5px", marginLeft: "10px" }}
          >
            <option value="">Select...</option>
            <option value="1">1 month</option>
            <option value="6">6 months</option>
            <option value="12">12 months</option>
          </select>
        </label>
      </div>
      
      <button 
        onClick={handleCalculate}
        style={{ 
          backgroundColor: "#4CAF50", 
          color: "white", 
          padding: "10px 15px", 
          border: "none", 
          borderRadius: "4px", 
          cursor: "pointer" 
        }}
      >
        Calculate
      </button>
      
      {result && (
        <div style={{ marginTop: "20px", padding: "15px", backgroundColor: "#f9f9f9", borderRadius: "5px" }}>
          <h3 style={{ marginTop: "0" }}>Results:</h3>
          <p><strong>Medication:</strong> {result.medication}</p>
          <p>
            <strong>Estimated weight change:</strong> 
            <span style={{ color: parseFloat(result.weightChange) >= 0 ? "#d32f2f" : "#388e3c" }}>
              {parseFloat(result.weightChange) >= 0 ? "+" : ""}{result.weightChange} {result.unit}
            </span>
          </p>
          <p><strong>Estimated new weight:</strong> {result.newWeight} {result.unit}</p>
          <p>
            <strong>Risk level:</strong> 
            <span style={{ 
              color: 
                result.riskLevel === "High" ? "#d32f2f" : 
                result.riskLevel === "Moderate" ? "#f57c00" : 
                result.riskLevel === "Low-Moderate" ? "#fbc02d" : 
                "#388e3c" 
            }}>
              {result.riskLevel}
            </span>
          </p>
        </div>
      )}
      
      <div style={{ marginTop: "20px", fontSize: "12px", color: "#666" }}>
        <p>This calculator is based on scientific research on antidepressant-associated weight changes.
        Individual results may vary. Always consult with a healthcare provider.</p>
      </div>
    </div>
  );
}
