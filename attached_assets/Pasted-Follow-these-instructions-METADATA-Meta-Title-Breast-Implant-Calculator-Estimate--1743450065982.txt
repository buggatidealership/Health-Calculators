        function calculate() {
            // Cup size to volume mapping (1 cup ≈ 150-200cc)
            const cupSizes = {
                "AA": 0, "A": 1, "B": 2, "C": 3,
                "D": 4, "DD/E": 5, "DDD/F": 6, "G": 7
            };
           
            // Get user inputs
            const currentCup = document.getElementById("currentCup").value;
            const goalCup = document.getElementById("goalCup").value;
            const breastWidth = parseFloat(document.getElementById("breastWidth").value);
            const implantType = document.getElementById("implantType").value;
            const location = document.getElementById("location").value;
           
            // Calculate volume needed (based on cup size difference)
            const cupDifference = cupSizes[goalCup] - cupSizes[currentCup];
            let volumeMin = cupDifference * 150;
            let volumeMax = cupDifference * 200;
           
            // Adjust for breast width (max implant width = breast width - 1cm)
            const maxImplantWidth = breastWidth - 1;
           
            // Determine profile suggestion based on width
            let profile = "Moderate";
            if (maxImplantWidth < 11) profile = "Moderate or Low";
            if (maxImplantWidth >= 13) profile = "High";
           
            // Calculate cost based on type and location
            let costPerImplantMin, costPerImplantMax;
           
            switch(implantType) {
                case "silicone":
                    costPerImplantMin = 1000;
                    costPerImplantMax = 2000;
                    break;
                case "saline":
                    costPerImplantMin = 600;
                    costPerImplantMax = 1500;
                    break;
                case "gummy":
                    costPerImplantMin = 1200;
                    costPerImplantMax = 2500;
                    break;
            }
           
            // Adjust for location
            let locationMultiplier = 1;
            if (location === "europe") locationMultiplier = 0.7;
            if (location === "asia") locationMultiplier = 0.5;
           
            // Surgeon fees (base)
            let surgeonFeeMin = 4000 * locationMultiplier;
            let surgeonFeeMax = 8000 * locationMultiplier;
           
            // Total cost range (with comma formatting)
            const totalMin = Math.round((costPerImplantMin * 2 + surgeonFeeMin) / 100) * 100;
            const totalMax = Math.round((costPerImplantMax * 2 + surgeonFeeMax) / 100) * 100;
           
            // Format numbers with commas
            const formatCost = (num) => num.toLocaleString('en-US');
           
            // Display results
            document.getElementById("volumeResult").textContent =
                `${volumeMin}-${volumeMax}cc (adjusted for your ${breastWidth}cm breast width)`;
            document.getElementById("profileResult").textContent =
                `${profile} profile (max width: ${maxImplantWidth}cm)`;
            document.getElementById("costResult").textContent =
                `$${formatCost(totalMin)}–$${formatCost(totalMax)} (${implantType} implants in ${location.toUpperCase()})`;
           
            document.getElementById("results").style.display = "block";
        }
