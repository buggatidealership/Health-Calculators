        document.addEventListener('DOMContentLoaded', function() {
            // Unit toggle functionality
            const metricBtn = document.getElementById('metric-btn');
            const imperialBtn = document.getElementById('imperial-btn');
            const metricInputs = document.getElementById('metric-inputs');
            const imperialInputs = document.getElementById('imperial-inputs');

                imperialInputs.style.display = 'block';
                metricInputs.style.display = 'none';
            });
           
            // Show/hide custom fasting input
            document.getElementById('fasting-type').addEventListener('change', function() {
                const customGroup = document.getElementById('custom-fasting-group');
                customGroup.style.display = this.value === 'custom' ? 'block' : 'none';

            // Calculate button
            document.getElementById('calculate').addEventListener('click', calculate);
           
            function calculate() {
                // Get unit system
                const isMetric = metricBtn.classList.contains('active');
               
                // Get weight and convert to kg if necessary
                let weightKg;
                if (isMetric) {
                    weightKg = parseFloat(document.getElementById('weight-kg').value);
                } else {
                    const weightLb = parseFloat(document.getElementById('weight-lb').value);
                    weightKg = weightLb * 0.453592;
                }
               
                // Get height and convert to cm if necessary
                let heightCm;
                if (isMetric) {
                    heightCm = parseFloat(document.getElementById('height-cm').value);
                } else {
                    const heightFt = parseFloat(document.getElementById('height-ft').value);
                    const heightIn = parseFloat(document.getElementById('height-in').value);
                    heightCm = (heightFt * 12 + heightIn) * 2.54;
                }
               
                // Get other inputs
                const age = parseInt(document.getElementById('age').value);
                const gender = document.getElementById('gender').value;
                const activity = parseFloat(document.getElementById('activity').value);
               
                let fastingHours = parseInt(document.getElementById('fasting-type').value);
                if (document.getElementById('fasting-type').value === 'custom') {
                    fastingHours = parseInt(document.getElementById('custom-fasting').value);
                }
               
                // Validate
                if (isNaN(weightKg) || isNaN(heightCm) || isNaN(age) || isNaN(fastingHours)) {
                    alert('Please fill in all fields with valid numbers');
                    return;
                }
               
                // Calculate BMR (Mifflin-St Jeor Equation)
                let bmr;
                if (gender === 'male') {
                    bmr = 10 * weightKg + 6.25 * heightCm - 5 * age + 5;
                } else {
                    bmr = 10 * weightKg + 6.25 * heightCm - 5 * age - 161;
                }
               
                // Calculate TDEE
                const tdee = bmr * activity;
               
                // Calculate fasting days
                const fastingDays = fastingHours / 24;
               
                // Calculate weight loss components
                // Based on research:
                // - First 24 hours: glycogen depletion + water loss (~0.5-1kg)
                // - After 24 hours: fat metabolism (~0.2-0.3kg fat per day)
                // - Minimal protein loss due to ketosis protein sparing effect
               
                const initialWaterLoss = Math.min(1, 0.7 + (0.3 * Math.random())); // 0.7-1kg
                const dailyFatLoss = 0.25; // kg per day after glycogen depletion
               
                let totalFatLoss, totalWaterLoss;
               
                if (fastingDays <= 1) {
                    // Short fasts are mostly water/glycogen
                    totalWaterLoss = initialWaterLoss * fastingDays;
                    totalFatLoss = 0.1 * fastingDays; // minimal fat loss
                } else {
                    // Longer fasts have initial water loss then fat loss
                    totalWaterLoss = initialWaterLoss;
                    totalFatLoss = dailyFatLoss * (fastingDays - 1);
                }
               
                const totalWeightLoss = totalFatLoss + totalWaterLoss;
               
                // Convert to pounds if using imperial
                let weightLossDisplay, fatLossDisplay, waterLossDisplay;
                if (isMetric) {
                    weightLossDisplay = totalWeightLoss.toFixed(2) + ' kg';
                    fatLossDisplay = totalFatLoss.toFixed(2) + ' kg';
                    waterLossDisplay = totalWaterLoss.toFixed(2) + ' kg';
                } else {
                    weightLossDisplay = (totalWeightLoss * 2.20462).toFixed(2) + ' lbs';
                    fatLossDisplay = (totalFatLoss * 2.20462).toFixed(2) + ' lbs';
                    waterLossDisplay = (totalWaterLoss * 2.20462).toFixed(2) + ' lbs';
                }
               
                // Display results
                document.getElementById('bmr').textContent = Math.round(bmr);
                document.getElementById('tdee').textContent = Math.round(tdee);
                document.getElementById('weight-loss').textContent = weightLossDisplay;
                document.getElementById('fat-loss').textContent = fatLossDisplay;
                document.getElementById('water-loss').textContent = waterLossDisplay;
               
                // Show results
                document.getElementById('results').style.display = 'block';
               
                // Create chart
                createChart(totalFatLoss, totalWaterLoss, isMetric);
            }
           
            function createChart(fatLoss, waterLoss, isMetric) {
                const ctx = document.getElementById('results-chart').getContext('2d');
               
                // Destroy previous chart if exists
                if (window.resultsChart) {
                    window.resultsChart.destroy();
                }
               
                const unit = isMetric ? 'kg' : 'lbs';
                const fatValue = isMetric ? fatLoss : fatLoss * 2.20462;
                const waterValue = isMetric ? waterLoss : waterLoss * 2.20462;
               
                window.resultsChart = new Chart(ctx, {
                    type: 'pie',
                    data: {
                        labels: [`Fat Loss (${fatValue.toFixed(2)} ${unit})`, `Water Weight Loss (${waterValue.toFixed(2)} ${unit})`],
                        datasets: [{
                            data: [fatLoss, waterLoss],
                            backgroundColor: [
                                '#3498db',
                                '#e74c3c'
                            ],
                            borderWidth: 1
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: {
                            legend: {
                                position: 'bottom'
                            },
                            tooltip: {
                                callbacks: {
                                    label: function(context) {
                                        const label = context.label || '';
                                        const value = context.raw || 0;
                                        const total = context.dataset.data.reduce((a, b) => a + b, 0);
                                        const percentage = Math.round((value / total) * 100);
                                        const displayValue = isMetric ? value.toFixed(2) : (value * 2.20462).toFixed(2);
                                        const displayUnit = isMetric ? 'kg' : 'lbs';
                                        return `${label}: ${displayValue} ${displayUnit} (${percentage}%)`;
                                    }
                                }
                            }
                        }
                    }
                });
            }
