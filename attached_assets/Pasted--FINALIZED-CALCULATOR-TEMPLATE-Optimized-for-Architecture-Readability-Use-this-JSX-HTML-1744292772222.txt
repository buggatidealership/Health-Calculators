import { BarChart, HelpCircle } from "lucide-react";
import { AdUnit } from "@/components/AdUnit";

export default function OzempicPenClickCalculator() {
  return (
    <div className="calculator-container">
      <h1>Ozempic Pen Click Calculator</h1>
      <p className="subtitle">Convert Ozempic pen clicks to milligrams of medication.</p>

      {/* ✅ Intro paragraph — plain English, 2–3 sentences max */}
      <p style={{ fontSize: "0.95rem", color: "#555", marginTop: 12 }}>
        This tool estimates your Ozempic dose by counting pen clicks. Helpful for understanding how much medication you're injecting or how much is left in your pen.
      </p>

      {/* ✅ Shared medical disclaimer — no inline variants allowed */}
      {% include 'components/medical-disclaimer.html' %}

      <AdUnit slot="ozempic-pen-click-calculator-top" />

      <div className="table-of-contents">
        <h3>Contents</h3>
        <ul>
          <li><a href="#calculator">Calculator</a></li>
          <li><a href="#how-to-use">How This Calculator Works</a></li>
          <li><a href="#faq">FAQ</a></li>
        </ul>
      </div>

      <section id="calculator">
        <div className="calculator">
          {/* BEGIN CLAUDE-GENERATED OZEMPIC PEN CLICK CALCULATOR */}
 import { useState } from 'react';
import { Calendar, Info, AlertTriangle, CheckCircle, Shield } from 'lucide-react';

export default function OzempicCalculator() {
  const [penStrength, setPenStrength] = useState("0.25-0.5");
  const [currentDose, setCurrentDose] = useState(0.25);
  const [weeksAtCurrentDose, setWeeksAtCurrentDose] = useState(4);
  const [lastInjectionDate, setLastInjectionDate] = useState("");
  const [showResults, setShowResults] = useState(false);
  const [safetyWarnings, setSafetyWarnings] = useState([]);

  // Pen specifications based on official Ozempic documentation
  const penSpecs = {
    "0.25-0.5": {
      clicks: 4,
      dosesPerClick: 0.25,
      maxDose: 2,
      penColor: "Blue",
      recommendedDoses: [0.25, 0.5]
    },
    "1-2": {
      clicks: 4,
      dosesPerClick: 0.5,
      maxDose: 4,
      penColor: "Gray",
      recommendedDoses: [1, 1.5, 2]
    },
    "2-4": {
      clicks: 4,
      dosesPerClick: 1,
      maxDose: 8,
      penColor: "Green",
      recommendedDoses: [2, 3, 4]
    }
  };

  // Calculate clicks needed for current dose
  const calculateClicks = () => {
    const spec = penSpecs[penStrength];
    return currentDose / spec.dosesPerClick;
  };

  // Calculate remaining doses in pen
  const calculateRemainingDoses = () => {
    const spec = penSpecs[penStrength];
    const totalClicksInPen = spec.clicks * (spec.maxDose / spec.dosesPerClick);
    const clicksUsedPerDose = calculateClicks();
    return Math.floor(totalClicksInPen / clicksUsedPerDose);
  };

  // Calculate next injection date
  const calculateNextInjectionDate = () => {
    if (!lastInjectionDate) return "Please enter your last injection date";
    
    const date = new Date(lastInjectionDate);
    date.setDate(date.getDate() + 7); // Ozempic is typically administered once weekly
    return date.toLocaleDateString();
  };

  // Calculate days since last injection
  const calculateDaysSinceLastInjection = () => {
    if (!lastInjectionDate) return null;
    
    const today = new Date();
    const lastInjection = new Date(lastInjectionDate);
    const diffTime = today - lastInjection;
    const diffDays = Math.floor(diffTime / (1000 * 60 * 60 * 24));
    return diffDays;
  };

  // Check if the dose is non-standard for the pen
  const isNonStandardDose = () => {
    const spec = penSpecs[penStrength];
    return !spec.recommendedDoses.includes(currentDose);
  };

  // Check dosing safety and generate warnings
  const checkDosingSafety = () => {
    const warnings = [];
    
    // Check if days since last injection is too short (potential overdose risk)
    const daysSince = calculateDaysSinceLastInjection();
    if (daysSince !== null) {
      if (daysSince < 6) {
        warnings.push(`It's only been ${daysSince} days since your last injection. Ozempic is typically taken once weekly (every 7 days).`);
      } else if (daysSince > 10) {
        warnings.push(`It's been ${daysSince} days since your last injection. Taking doses too far apart may affect medication effectiveness.`);
      }
    }
    
    // Check if the dose is not standard for the selected pen
    if (isNonStandardDose()) {
      warnings.push(`The dose of ${currentDose}mg is not a standard dose for the ${penSpecs[penStrength].penColor} pen. This increases risk of dosing errors.`);
    }
    
    // Check if clicks are not a whole number (risk of inaccurate dosing)
    const clicks = calculateClicks();
    if (clicks % 1 !== 0) {
      warnings.push(`Your dose requires ${clicks} clicks, which is not a whole number. This may lead to inaccurate dosing.`);
    }
    
    return warnings;
  };

  // Handle button click directly
  const handleCalculateClick = () => {
    const warnings = checkDosingSafety();
    setSafetyWarnings(warnings);
    setShowResults(true);
  };

  return (
    <div className="max-w-md mx-auto p-6 bg-white rounded-lg shadow-lg">
      <div className="bg-red-50 border-l-4 border-red-500 p-4 mb-6">
        <div className="flex items-start">
          <AlertTriangle className="w-5 h-5 text-red-500 mt-1 mr-2" />
          <div>
            <p className="font-bold text-red-700">Medical Disclaimer</p>
            <p className="text-sm text-red-700">
              This calculator is for informational purposes only and does not replace medical advice. 
              Always follow your healthcare provider's specific instructions regarding dosing, 
              frequency, and any changes to your medication regimen.
            </p>
          </div>
        </div>
      </div>
      
      <h1 className="text-2xl font-bold text-center mb-6 text-blue-600">Ozempic Dosing Safety Calculator</h1>
      
      <div className="space-y-4">
        <div className="space-y-2">
          <label className="block text-sm font-medium text-gray-700">Pen Strength (mg)</label>
          <select
            value={penStrength}
            onChange={(e) => setPenStrength(e.target.value)}
            className="w-full p-2 border rounded-md"
          >
            <option value="0.25-0.5">0.25-0.5 mg (Blue)</option>
            <option value="1-2">1-2 mg (Gray)</option>
            <option value="2-4">2-4 mg (Green)</option>
          </select>
        </div>
        
        <div className="space-y-2">
          <label className="block text-sm font-medium text-gray-700">Current Dose (mg)</label>
          <select
            value={currentDose}
            onChange={(e) => setCurrentDose(parseFloat(e.target.value))}
            className="w-full p-2 border rounded-md"
          >
            {penStrength === "0.25-0.5" && (
              <>
                <option value="0.25">0.25 mg</option>
                <option value="0.5">0.5 mg</option>
                <option value="0.75">0.75 mg (non-standard)</option>
              </>
            )}
            {penStrength === "1-2" && (
              <>
                <option value="1">1 mg</option>
                <option value="1.5">1.5 mg</option>
                <option value="2">2 mg</option>
                <option value="2.5">2.5 mg (non-standard)</option>
              </>
            )}
            {penStrength === "2-4" && (
              <>
                <option value="2">2 mg</option>
                <option value="3">3 mg</option>
                <option value="4">4 mg</option>
                <option value="5">5 mg (non-standard)</option>
              </>
            )}
          </select>
          <p className="text-xs text-gray-500">Standard doses are recommended to prevent dosing errors</p>
        </div>
        
        <div className="space-y-2">
          <label className="block text-sm font-medium text-gray-700">Weeks at Current Dose</label>
          <input
            type="number"
            min="0"
            value={weeksAtCurrentDose}
            onChange={(e) => setWeeksAtCurrentDose(parseInt(e.target.value))}
            className="w-full p-2 border rounded-md"
          />
        </div>
        
        <div className="space-y-2">
          <label className="block text-sm font-medium text-gray-700">Last Injection Date</label>
          <input
            type="date"
            value={lastInjectionDate}
            onChange={(e) => setLastInjectionDate(e.target.value)}
            className="w-full p-2 border rounded-md"
          />
        </div>
        
        <button 
          type="button" 
          onClick={handleCalculateClick}
          className="w-full py-2 px-4 bg-blue-600 text-white font-semibold rounded-md hover:bg-blue-700"
        >
          Calculate & Check Safety
        </button>
      </div>
      
      {showResults && (
        <div className="mt-8">
          {safetyWarnings.length > 0 && (
            <div className="mb-6 p-4 bg-orange-50 border-l-4 border-orange-500 rounded">
              <h3 className="flex items-center text-orange-800 font-bold">
                <Shield className="w-5 h-5 mr-2" />
                Safety Alerts
              </h3>
              <ul className="mt-2 space-y-2 text-sm text-orange-800">
                {safetyWarnings.map((warning, index) => (
                  <li key={index} className="flex items-start">
                    <span className="mr-2">•</span>
                    <span>{warning}</span>
                  </li>
                ))}
              </ul>
            </div>
          )}
          
          <div className="p-4 bg-blue-50 rounded-md">
            <h2 className="text-xl font-semibold text-blue-800 mb-4">Dosing Information</h2>
            
            <div className="space-y-4">
              <div className="flex items-start space-x-2">
                <CheckCircle className="w-5 h-5 text-green-500 mt-1" />
                <div>
                  <p className="font-medium">Pen Clicks for Your Dose:</p>
                  <p className="text-lg font-bold">{calculateClicks()} clicks</p>
                  {calculateClicks() % 1 !== 0 && (
                    <p className="text-xs text-red-600">Fractional clicks may cause dosing errors</p>
                  )}
                </div>
              </div>
              
              <div className="flex items-start space-x-2">
                <Calendar className="w-5 h-5 text-blue-500 mt-1" />
                <div>
                  <p className="font-medium">Next Injection Date:</p>
                  <p className="text-lg font-bold">{calculateNextInjectionDate()}</p>
                  <p className="text-xs text-gray-600">Based on standard once-weekly administration</p>
                </div>
              </div>
              
              <div className="flex items-start space-x-2">
                <Info className="w-5 h-5 text-purple-500 mt-1" />
                <div>
                  <p className="font-medium">Remaining Doses in Pen:</p>
                  <p className="text-lg font-bold">Approximately {calculateRemainingDoses()} doses</p>
                </div>
              </div>

              {calculateDaysSinceLastInjection() !== null && (
                <div className="flex items-start space-x-2">
                  <Info className="w-5 h-5 text-blue-500 mt-1" />
                  <div>
                    <p className="font-medium">Days Since Last Injection:</p>
                    <p className="text-lg font-bold">{calculateDaysSinceLastInjection()} days</p>
                    {calculateDaysSinceLastInjection() < 7 && (
                      <p className="text-xs text-orange-600">Recommended interval is 7 days between injections</p>
                    )}
                  </div>
                </div>
              )}
            </div>
            
            <div className="mt-6 p-4 bg-green-50 border border-green-200 rounded">
              <h3 className="font-semibold text-green-800">Dosing Best Practices:</h3>
              <ul className="mt-2 space-y-2 text-sm">
                <li className="flex items-start">
                  <span className="mr-2">•</span>
                  <span>Use the correct pen for your prescribed dose range</span>
                </li>
                <li className="flex items-start">
                  <span className="mr-2">•</span>
                  <span>Count clicks carefully - each click must be complete until you hear/feel a "click"</span>
                </li>
                <li className="flex items-start">
                  <span className="mr-2">•</span>
                  <span>Maintain a consistent 7-day interval between injections</span>
                </li>
                <li className="flex items-start">
                  <span className="mr-2">•</span>
                  <span>If you miss a dose and it's more than 5 days until your next scheduled dose, take the missed dose as soon as possible</span>
                </li>
                <li className="flex items-start">
                  <span className="mr-2">•</span>
                  <span>Keep a log of injections with dates, doses, and any side effects</span>
                </li>
              </ul>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}

          {/* END CLAUDE-GENERATED OZEMPIC PEN CLICK CALCULATOR */}
        </div>
      </section>

      <AdUnit slot="ozempic-pen-click-calculator-mid" />

      {/* ✅ Optional Related Guide Block */}
      {false && (
        <div className="highlight" style={{ marginTop: 40, backgroundColor: '#f8f9fa', padding: 20, borderRadius: 8, borderLeft: '4px solid #d63384' }}>
          <h3 style={{ color: '#333', marginTop: 0 }}>📘 Related Guide</h3>
          <p>
            Learn how Ozempic pens are calibrated and how pen clicks correspond to injectable volumes.
            <a href="/resources/ozempic-pen-click-guide" className="pink-link">Read the full guide →</a>
          </p>
        </div>
      )}

      <section id="how-to-use">
        <h2><BarChart className="section-icon" /> How to Use the Ozempic Pen Click Calculator</h2>
        <ul>
          <li>Enter the number of pen clicks for your intended dose.</li>
          <li>Select the pen type (dose size in mg).</li>
          <li>See your calculated dose based on average click spacing.</li>
        </ul>
      </section>

      <section id="faq">
        <h2><HelpCircle className="section-icon" /> Frequently Asked Questions</h2>
        <div className="faq-item">
          <h4>How accurate are Ozempic pen clicks?</h4>
          <p>Each pen has a specific number of clicks per dose. Our calculator uses those specs for its estimates.</p>
        </div>
        <div className="faq-item">
          <h4>Can I use this with Wegovy or other GLP-1 pens?</h4>
          <p>No. This calculator is designed specifically for Ozempic pens. Other pens may have different click calibrations.</p>
        </div>
      </section>

      <AdUnit slot="ozempic-pen-click-calculator-bottom" />

      <footer className="footer-disclaimer">
        <p><a href="/">&larr; Back to Homepage</a></p>
        <p><strong>Disclaimer:</strong> This calculator is for informational purposes only and does not replace medical advice.</p>
      </footer>
    </div>
  );
}
