        function calculateCost() {
            // Base price per syringe (average from ASPS 2023 data)
            const basePricePerSyringe = 684;
           
            // Get user inputs
            const locationFactor = parseFloat(document.getElementById('location').value);
            const providerFactor = parseFloat(document.getElementById('provider').value);
            const productFactor = parseFloat(document.getElementById('product').value);
            const syringes = parseFloat(document.getElementById('syringes').value);
            const touchupFactor = parseFloat(document.getElementById('touchups').value);
           
            // Calculate cost
            let cost = basePricePerSyringe * locationFactor * providerFactor * productFactor * syringes;
            let touchupCost = cost * touchupFactor;
            let totalCost = cost + touchupCost;
           
            // Display result
            document.getElementById('price').textContent = '$' + Math.round(totalCost) + ' - $' + Math.round(totalCost * 1.2);
           
            let details = `Breakdown:
                <ul>
                    <li>Base cost for ${syringes} syringe(s): $${Math.round(cost)}</li>
                    ${touchupFactor > 0 ? `<li>Estimated touch-up costs: $${Math.round(touchupCost)}</li>` : ''}
                    <li>Price factors: ${getLocationText()} location, ${getProviderText()} provider, ${getProductText()} product</li>
                </ul>`;
           
            document.getElementById('details').innerHTML = details;
            document.getElementById('result').style.display = 'block';
        }
       
        function getLocationText() {
            const options = document.getElementById('location').options;
            return options[options.selectedIndex].text.split(' (')[0];
        }
       
        function getProviderText() {
            const options = document.getElementById('provider').options;
            return options[options.selectedIndex].text.split(' (')[0];
        }
       
        function getProductText() {
            const options = document.getElementById('product').options;
            return options[options.selectedIndex].text.split(' (')[0];
        }

Q: How much lip filler do I need?
A: Most first-time patients get 0.5 to 1.0ml. 1.5–2.0ml may be used for fuller lips or touch-ups.

A: Average cost ranges from $500–$850 per ml depending on brand and provider.

A: Yes. Nurses typically charge less than board-certified dermatologists or surgeons.
