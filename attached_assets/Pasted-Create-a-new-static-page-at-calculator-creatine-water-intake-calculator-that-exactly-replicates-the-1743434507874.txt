        function calculateWater() {
            // Get input values
            const weight = parseFloat(document.getElementById('weight').value);
            const phase = document.getElementById('phase').value;
            const activity = document.getElementById('activity').value;
           
            // Validate weight
            if (isNaN(weight) || weight < 30 || weight > 200) {
                alert('Please enter a valid weight between 30 and 200 kg');
                return;
            }
           
            // Calculate base water needs (30-35 mL per kg of body weight)
            let baseWater = weight * 35;
           
            // Adjust for activity level
            let activityMultiplier = 1;
            if (activity === 'high') {
                activityMultiplier = 1.2;
            } else if (activity === 'extreme') {
                activityMultiplier = 1.4;
            }
           
            // Calculate total water without creatine adjustment
            let totalWater = baseWater * activityMultiplier;
           
            // Add creatine adjustment (additional 500-1000 mL based on phase)
            let creatineAdjustment = 0;
            let creatineDose = '';
           
            if (phase === 'loading') {
                creatineAdjustment = 1000;
                creatineDose = 'During the loading phase (first 5-7 days), take 20g of creatine monohydrate daily (divided into 4 doses of 5g each).';
                totalWater += creatineAdjustment;
            } else {
                creatineAdjustment = 500;
                creatineDose = 'During the maintenance phase, take 3-5g of creatine monohydrate daily.';
                totalWater += creatineAdjustment;
            }
           
            // Convert to liters and round
            const totalWaterLiters = Math.round(totalWater / 100) / 10;
            const baseWaterLiters = Math.round((baseWater * activityMultiplier) / 100) / 10;
