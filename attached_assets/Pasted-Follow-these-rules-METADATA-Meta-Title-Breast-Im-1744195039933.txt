                    weightInput.value = weightInput.value ? (parseFloat(weightInput.value) * kgToLbs).toFixed(1) : '';
                } else {
                    chestInput.value = (parseFloat(chestInput.value) * inchToCm).toFixed(1);
                    bustInput.value = bustInput.value ? (parseFloat(bustInput.value) * inchToCm).toFixed(1) : '';
                    heightInput.value = heightInput.value ? (parseFloat(heightInput.value) * inchToCm).toFixed(1) : '';

                case 'medium': frameFactor = 1.0; break;
                case 'large': frameFactor = 1.2; break;
            }
            
            // Adjust for body weight and height (BMI factor)
            const heightM = heightCm / 100;
            const bmi = weightKg / (heightM * heightM);
            let bmiFactor = 1.0;
            
            if (bmi < 18.5) {
                bmiFactor = 0.85; // Underweight
            } else if (bmi >= 25) {
                bmiFactor = 1.15; // Overweight
            }
            
            // Final recommended implant size (range)
            const adjustedImplantSize = baseImplantSize * frameFactor * bmiFactor;
            const lowerRange = Math.round(adjustedImplantSize * 0.85);
            const upperRange = Math.round(adjustedImplantSize * 1.15);
            
            // Calculate projected cup size
            const projectVolume = currentVolume + adjustedImplantSize;
            let projectedCup = '';

                projectedCup = 'AA';

                projectedCup = 'A';

                projectedCup = 'C';

                projectedCup = 'D';

                projectedCup = 'DD/E';
            } else {
                projectedCup = 'F or larger';
