    <script>
        document.addEventListener('DOMContentLoaded', function() {
            document.getElementById('calculate-btn').addEventListener('click', calculate);
        });

        function calculate() {
            // Get input values
            const weight = parseFloat(document.getElementById('weight').value);
            const height = parseFloat(document.getElementById('height').value);
            const gender = document.getElementById('gender').value;
            const location = document.getElementById('location').value;
           
            // Get selected units
            const weightUnit = document.querySelector('input[name="weight-unit"]:checked').value;
            const heightUnit = document.querySelector('input[name="height-unit"]:checked').value;
           
            // Get selected areas
            const selectedAreas = [];
            document.querySelectorAll('input[name="area"]:checked').forEach(checkbox => {
                selectedAreas.push(checkbox.value);
            });
            const areasCount = selectedAreas.length;
           
            // Validate inputs
            if (isNaN(weight)) {
                alert("Please enter your current weight");
                return;
            }
            if (isNaN(height)) {
                alert("Please enter your height");
                return;
            }
            if (areasCount === 0) {
                alert("Please select at least one treatment area");
                return;
            }
           
            // Convert to standard units (lbs and inches)
            let weightLbs = weight;
            if (weightUnit === 'kg') {
                weightLbs = weight * 2.20462;
            }
           
            let heightInches = height;
            if (heightUnit === 'cm') {
                heightInches = height / 2.54;
            }
           
            // Calculate BMI
            const bmi = (weightLbs / (heightInches * heightInches)) * 703;
            let bmiCategory = "";
            if (bmi < 18.5) bmiCategory = "Underweight";
            else if (bmi < 25) bmiCategory = "Normal weight";
            else if (bmi < 30) bmiCategory = "Overweight";
            else bmiCategory = "Obese";
           
            // Calculate fat removal (based on safety guidelines and typical amounts)
            let maxFatLiters = 5; // Absolute maximum for outpatient
            let typicalFatLiters = 0;
           
            if (bmi < 25) {
                // Normal weight patients - typically remove 1-3 liters
                typicalFatLiters = 1 + (areasCount * 0.5);
                if (typicalFatLiters > 3) typicalFatLiters = 3;
            } else if (bmi < 30) {
                // Overweight patients - typically remove 2-4 liters
                typicalFatLiters = 2 + (areasCount * 0.6);
                if (typicalFatLiters > 4) typicalFatLiters = 4;
            } else {
                // Obese patients - may remove up to 5 liters but with higher risk
                typicalFatLiters = 3 + (areasCount * 0.4);
                if (typicalFatLiters > 5) typicalFatLiters = 5;
            }
           
            // Convert liters to pounds (fat density ~0.9 kg/L, 2.2 lbs/kg)
            const fatRemovalLbs = typicalFatLiters * 0.9 * 2.2;
           
            // Calculate new weight
            const newWeightLbs = weightLbs - fatRemovalLbs;
           
            // Convert back to original unit if needed
            let newWeightDisplay = newWeightLbs.toFixed(1) + ' lbs';
            if (weightUnit === 'kg') {
                newWeightDisplay = (newWeightLbs / 2.20462).toFixed(1) + ' kg';
            }
           
            // Calculate cost
            let costPerArea = 0;
            switch(location) {
                case 'us':
                    costPerArea = 2500;
                    break;
                case 'uk':
                    costPerArea = 2000;
                    break;
                case 'eu':
                    costPerArea = 1800;
                    break;
                case 'asia':
                    costPerArea = 1500;
                    break;
                default:
                    costPerArea = 2000;
            }
           
            // Apply gender adjustment (males often pay slightly more)
            if (gender === 'male') {
                costPerArea *= 1.1;
            }
           
            // Calculate total cost with area multiplier
            let totalCost = areasCount * costPerArea;
            if (areasCount >= 5) {
                totalCost *= 0.9; // Discount for multiple areas
            }
           
            // Format cost
            const costFormatted = new Intl.NumberFormat('en-US', {
                style: 'currency',
                currency: 'USD'
            }).format(totalCost);
           
            // Format area names
            const areaNames = {
                'abdomen': 'Abdomen',
                'flanks': 'Flanks',
                'thighs': 'Thighs',
                'arms': 'Arms',
                'back': 'Back',
                'chin': 'Chin/Neck',
                'chest': 'Chest',
                'buttocks': 'Buttocks'
            };
            const formattedAreas = selectedAreas.map(area => areaNames[area]).join(', ');
           
            // Display results
            document.getElementById('bmi').textContent = bmi.toFixed(1);
            document.getElementById('bmi-category').textContent = bmiCategory;
            document.getElementById('selected-areas').textContent = formattedAreas;
            document.getElementById('fat-removal').textContent =
                `${typicalFatLiters.toFixed(1)} liters (≈${fatRemovalLbs.toFixed(1)} lbs)`;
            document.getElementById('new-weight').textContent = newWeightDisplay;
            document.getElementById('cost').textContent =
                `${costFormatted} (varies by surgeon, facility, and anesthesia)`;
           
            document.getElementById('results').style.display = 'block';
        }
    </script>
