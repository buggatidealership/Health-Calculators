        function calculateVitaminD() {
            // Get user inputs
            const age = parseInt(document.getElementById('age').value);
            const pregnant = document.getElementById('pregnant').checked;
            const skin = parseInt(document.getElementById('skin').value);
            const sun = document.getElementById('sun').value;
            const level = document.getElementById('level').value;
           
            // Validate age
            if (isNaN(age) || age < 0 || age > 120) {
                alert("Please enter a valid age between 0 and 120");
                return;
            }
           
            // Base recommendations by exact age (IU/day)
            let baseRecommendation;
            if (pregnant) {
                baseRecommendation = 1500; // Special recommendation for pregnancy/breastfeeding
            } else if (age < 1) {
                baseRecommendation = 400; // Infants
            } else if (age <= 18) {
                // Gradual increase from 600 to 1000 IU for children
                baseRecommendation = 600 + Math.min(400, (age - 1) * 23.5);
            } else if (age <= 70) {
                // Adults 19-70
                baseRecommendation = 1500;
            } else {
                // Seniors 70+, gradual increase from 2000 to 3000 IU with age
                baseRecommendation = 2000 + Math.min(1000, (age - 70) * 20);
            }
           
            // Adjust for skin type (darker skin needs more)
            const skinAdjustment = (skin - 3) * 200; // Type 3 is baseline
           
            // Adjust for sun exposure
            let sunAdjustment;
            switch(sun) {
                case 'low': sunAdjustment = 0; break;
                case 'moderate': sunAdjustment = -400; break;
                case 'high': sunAdjustment = -800; break;
                default: sunAdjustment = 0;
            }
           
            // Adjust if blood level is known
            let levelAdjustment = 0;
            let levelMessage = '';
            if (level) {
                const levelNum = parseFloat(level);
                if (levelNum < 20) {
                    levelAdjustment = 3000; // Deficiency - therapeutic dose
                    levelMessage = `Your level of ${level} ng/mL indicates deficiency. `;
                } else if (levelNum < 30) {
                    levelAdjustment = 1000; // Insufficiency
                    levelMessage = `Your level of ${level} ng/mL indicates insufficiency. `;
                } else if (levelNum > 50) {
                    levelAdjustment = -500; // High level
                    levelMessage = `Your level of ${level} ng/mL is above optimal range. `;
                }
            }
           
            // Calculate total recommendation
            let total = baseRecommendation + skinAdjustment + sunAdjustment + levelAdjustment;
           
            // Ensure minimum of 400 IU and maximum of 4000 IU (upper limit per IOM)
            total = Math.max(400, Math.min(total, 4000));
           
            // Generate result message
            let resultHTML = `<h3>Recommended Vitamin D Intake</h3>`;
            resultHTML += `<p><strong>${Math.round(total)} IU per day</strong></p>`;
           
            if (levelMessage) {
                resultHTML += `<p>${levelMessage}</p>`;
            }
           
            if (total >= 2000) {
                resultHTML += `<p>This higher dose is recommended for a limited time (8-12 weeks) to correct deficiency. After this period, retest your levels and adjust dosage accordingly.</p>`;
            }
           
            resultHTML += `<p>Remember to get your Vitamin D levels checked regularly, especially when taking supplements above 2000 IU/day.</p>`;
           
            // Display result
            const resultDiv = document.getElementById('result');
            resultDiv.innerHTML = resultHTML;
            resultDiv.style.display = 'block';
        }
